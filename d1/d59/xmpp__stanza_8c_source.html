00359 static void parse_ns(const char *ns_name, char **name, char **prefix,
00360                      char **uri) {
00361     char *separator = strchr(ns_name, XMPP_PARSER_SEPARATOR);
00362     if (separator == NULL) {
00363         /* No namespace or prefix. */
00364         STRDUP_CHECK(*name, ns_name);
00365 
00366     } else {
00367         /* There is a namespace URI. */
00368         STRNDUP_CHECK(*uri, ns_name, separator - ns_name);
00369 
00370         char *tmp = separator + 1;
00371         separator = strchr(tmp, XMPP_PARSER_SEPARATOR);
00372         if (separator == NULL) {
00373             /* There is no namespace prefix. */
00374             STRDUP_CHECK(*name, tmp);
00375         } else {
00376             /* There is a namespace prefix. */
00377             STRNDUP_CHECK(*name, tmp, separator - tmp);
00378             STRDUP_CHECK(*prefix, separator + 1);
00379         }
00380     }

00389     if (stanza->prefix) {
00390         utstring_printf(str, "<%s:%s", stanza->prefix, stanza->name);
00391     } else {
00392         utstring_printf(str, "<%s", stanza->name);
00393     }
00394 
00395     struct xmpp_parser_namespace *ns = stanza->namespaces;
00396     while (ns != NULL) {
00397         const char *prefix = xmpp_parser_namespace_prefix(ns);
00398         const char *uri = xmpp_parser_namespace_uri(ns);
00399         if (prefix) {
00400             utstring_printf(str, " xmlns:%s='%s'", prefix, uri);
00401         } else {
00402             utstring_printf(str, " xmlns='%s'", uri);
00403         }
00404         ns = xmpp_parser_namespace_next(ns);
00405     }
00406 
00407     struct attribute *attr, *tmp;
00408     HASH_ITER(hh, stanza->attributes, attr, tmp) {
00409         char quot;
00410         if (strchr(attr->value, '\'') != NULL) {
00411             quot = '"';
00412         } else {
00413             quot = '\'';
00414         }
00415         if (attr->prefix) {
00416             utstring_printf(str, " %s:%s=%c%s%c", attr->prefix, attr->name,
00417                             quot, attr->value, quot);
00418         } else {
00419             utstring_printf(str, " %s=%c%s%c", attr->name, quot, attr->value,
00420                             quot);
00421         }
00422     }
00423 
00424     if (stanza->children != NULL || utstring_len(&stanza->data) > 0) {
00425         utstring_printf(str, ">");
00426         utstring_concat(str, &stanza->data);
00427         struct xmpp_stanza *child;
00428         DL_FOREACH(stanza->children, child) {
00429             stanza_tostr(child, str);
00430         }
00431         if (stanza->prefix) {
00432             utstring_printf(str, "</%s:%s>", stanza->prefix, stanza->name);
00433         } else {
00434             utstring_printf(str, "</%s>", stanza->name);
00435         }
00436     } else {
