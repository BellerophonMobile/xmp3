00241 static int ini_handler(void *data, const char *section, const char *name,
00242                         const char *value) {
00243     struct xmp3_options *options = data;
00244 
00245     /* The top of the file (no section) is for configuring core XMP3. */
00246     if (strcmp(section, "") == 0) {
00247         // Main XMP3 options
00248         if (strcmp(name, "address") == 0) {
00249             return xmp3_options_set_addr_str(options, value);
00250         }
00251 
00252         if (strcmp(name, "port") == 0) {
00253             return xmp3_options_set_port_str(options, value);
00254         }
00255 
00256         if (strcmp(name, "ssl") == 0) {
00257             if (strcmp(value, "true") == 0) {
00258                 return xmp3_options_set_ssl(options, true);
00259             } else if (strcmp(value, "false") == 0) {
00260                 return xmp3_options_set_ssl(options, false);
00261             } else {
00262                 log_err("Invalid value for ssl option: '%s'", value);
00263                 return false;
00264             }
00265         }
00266 
00267         if (strcmp(name, "keyfile") == 0) {
00268             return xmp3_options_set_keyfile(options, value);
00269         }
00270 
00271         if (strcmp(name, "certificate") == 0) {
00272             return xmp3_options_set_certificate(options, value);
00273         }
00274 
00275         if (strcmp(name, "name") == 0) {
00276             return xmp3_options_set_server_name(options, value);
00277         }
00278 
00279         if (strcmp(name, "modpath") == 0) {
00280             return xmp3_options_add_module_path(options, value);
00281         }
00282 
00283         log_err("Unknown config item '%s = %s'", name, value);
00284         return false;
00285 
00286     /* The "modules" section is for loading modules.  Modules get assigned
00287      * to "name".  This name can then be used as sections to specify options
00288      * to the module. */
00289     } else if (strcmp(section, "modules") == 0) {
00290         char path[PATH_MAX];
00291         tj_searchpathlist_locate(options->search_path, value, path, PATH_MAX);
00292         if (!xmp3_modules_load(options->modules, path, name)) {
00293             log_err("Error loading module '%s' (%s)", path, value);
00294             return false;
00295         }
00296 
00297     /* Any other section is used to configure a module loaded in the "modules"
00298      * section. */
00299     } else {
00300         if (!xmp3_modules_config(options->modules, section, name, value)) {
00301             log_err("Error configuring module '%s'", section);
00302             return false;
00303         }
00304     }
00305     return true;
00306 }
