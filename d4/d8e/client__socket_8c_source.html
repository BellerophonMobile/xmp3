00155     size_t numsent = 0;

00157         ssize_t newsent = client_socket_send(socket, buf + numsent,

00218 static void ssl_close(struct client_socket *socket) {
00219     struct ssl_socket *self = (struct ssl_socket*)socket->self;
00220 
00221     /* If the socket has already been closed, SSL_shutdown will cause
00222      * a SIGPIPE to be sent to this process.  We don't care, so block it
00223      * every time. */
00224     struct sigaction sa = { .sa_handler=SIG_IGN };
00225     struct sigaction osa;
00226     if (sigaction(SIGPIPE, &sa, &osa) == 0) {
00227         if (SSL_shutdown(self->ssl) != 1) {
00228             ERR_print_errors_fp(stderr);
00229         }
00230         if (sigaction(SIGPIPE, &osa, NULL) != 0) {
00231             log_err("Can't unignore SIGPIPE");
00232         }
00233     } else {
00234         log_err("Can't ignore SIGPIPE.");
00235     }
00236 
00237     socket->self = self->fd_socket;
00238     fd_close(socket);
00239     socket->self = self;
00240 }
