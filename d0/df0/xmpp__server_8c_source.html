00410 void xmpp_server_disconnect_client(struct xmpp_client *client) {
00411     struct xmpp_server *server = xmpp_client_server(client);
00412 
00413     /* Sanity check that the client is registered with the server. */
00414     struct c_client *search = NULL;
00415     DL_FOREACH(server->clients, search) {
00416         if (client == search->client) {
00417             break;
00418         }
00419     }
00420     if (search == NULL) {
00421         log_warn("Attempted to disconnect non-registered client.");
00422         return;
00423     }
00424 
00425     DL_DELETE(server->clients, search);
00426 
00427     struct client_listener *listener = NULL;
00428     struct client_listener *tmp = NULL;
00429     DL_FOREACH_SAFE(server->client_listeners, listener, tmp) {
00430         if (listener->client == client) {
00431             listener->cb(client, listener->data);
00432             DL_DELETE(server->client_listeners, listener);
00433             free(listener);
00434         }
00435     }
00436

00499         strjid = jid_to_str(route->jid);

00501         free(strjid);
00502         if (jid_cmp_wildcards(search_jid, route->jid) == 0) {
00503             debug("Yes");
00504             if (route->cb(stanza, server, route->data)) {
00505                 was_handled = true;
00506             }

00508             debug("No");
00509         }
00510     }
00511     if (!was_handled) {
