00171             if (errno == EINTR) {
00172                 break;

00174                 log_err("Event loop select error");
00175                 return;

00177         }
00178 
00179         if (num_ready == 0) {
00180             /* If/when we do timeout-based callbacks, it would happen here. */
00181             continue;
00182         }
00183 
00184         struct event_item *item;
00185         struct event_item *item_tmp;
00186         DL_FOREACH_SAFE(loop->events, item, item_tmp) {
00187             if (FD_ISSET(item->fd, &loop->readfds)) {
00188                 item->func(loop, item->fd, item->data);
00189             }
