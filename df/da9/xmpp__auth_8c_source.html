00288 static bool handle_sasl_plain(struct xmpp_stanza *stanza,
00289                               struct xmpp_parser *parser, void *data) {
00290     struct xmpp_client *client = (struct xmpp_client*)data;
00291     char *plaintext = NULL;
00292 
00293     debug("SASL plain authentication");
00294 
00295     check(strcmp(xmpp_stanza_uri(stanza), SASL_NS) == 0,
00296           "Unexpected stanza");
00297     check(strcmp(xmpp_stanza_name(stanza), AUTH) == 0,
00298           "Unexpected stanza");
00299 
00300     const char *mechanism = xmpp_stanza_attr(stanza, AUTH_MECHANISM);
00301     check(mechanism != NULL && strcmp(mechanism, AUTH_MECHANISM_PLAIN) == 0,
00302           "Unexpected authentication mechanism.");
00303 
00304     check(xmpp_stanza_data_length(stanza) * 0.75 <= PLAIN_AUTH_BUFFER_SIZE,
00305           "Auth plaintext buffer overflow");
00306 
00307     plaintext = base64_decode(xmpp_stanza_data(stanza),
00308                               xmpp_stanza_data_length(stanza));
00309 
00310     char *authzid = plaintext;
00311     char *authcid = memchr(authzid, '\0', PLAIN_AUTH_BUFFER_SIZE) + 1;
00312     char *passwd = memchr(authcid, '\0', PLAIN_AUTH_BUFFER_SIZE) + 1;
00313 
00314     debug("authzid = '%s', authcid = '%s', passwd = '%s'",
00315           authzid, authcid, passwd);
00316 
00317     if (!xmpp_server_authenticate(xmpp_client_server(client), authzid, authcid,
00318                                   passwd)) {
00319         log_warn("User '%s' failed to authenticate, disconnecting.", authcid);
00320         check(client_socket_sendall(xmpp_client_socket(client),
00321                                     MSG_NOT_AUTHORIZED, 
00322                                     strlen(MSG_NOT_AUTHORIZED)) > 0,
00323               "Error sending not authorized message to client");
00324         goto error;
00325     }
00326 
00327     struct jid *jid = jid_new();
00328     jid_set_local(jid, authcid);
00329     jid_set_domain(jid,
00330             jid_domain(xmpp_server_jid(xmpp_client_server(client))));
00331     xmpp_client_set_jid(client, jid);
00332 
00333     /* Success! */
00334     check(client_socket_sendall(xmpp_client_socket(client),
00335                 MSG_SASL_SUCCESS, strlen(MSG_SASL_SUCCESS)) > 0,
00336           "Error sending SASL success to client");
00337 
00338     /* Go to step 7, the client needs to send us a new stream header. */
00339     xmpp_parser_new_stream(parser);
00340     xmpp_parser_set_handler(parser, stream_bind_start);
00341 
00342     char *strjid = jid_to_str(jid);
00343     log_info("User %s connected.", strjid);
00344     free(strjid);
00345 
00346     free(plaintext);
00347     return true;
00348 
00349 error:
00350     if (plaintext != NULL) {
00351         free(plaintext);
00352     }
00353     return false;

00359 static bool handle_bind_iq(struct xmpp_stanza *stanza,
00360                            struct xmpp_parser *parser, void *data) {
00361     struct xmpp_client *client = (struct xmpp_client*)data;
00362     struct xmpp_server *server = xmpp_client_server(client);

00364     debug("Resource binding IQ");
00365 
00366     UT_string success_msg;
00367     utstring_init(&success_msg);

00369     check(strcmp(xmpp_stanza_uri(stanza), XMPP_STANZA_NS_CLIENT) == 0,
00370           "Unexpected stanza");
00371     check(strcmp(xmpp_stanza_name(stanza), XMPP_STANZA_IQ) == 0,
00372           "Unexpected stanza");
00373 
00374     /* Validate the correct attributes set on the start tag. */
00375     const char *type = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TYPE);
00376     check(type != NULL && strcmp(type, XMPP_STANZA_TYPE_SET) == 0,
00377           "Unexpected bind iq type.");
00378 
00379     const char *id = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_ID);
00380     check(id != NULL, "Bind iq has no id.");
00381 
00382     /* Jump to the inner <bind> tag. */
00383     stanza = xmpp_stanza_children(stanza);
00384     check(stanza != NULL, "Bind iq has no child.");
00385     check(strcmp(xmpp_stanza_uri(stanza), BIND_NS) == 0,
00386           "Unexpected stanza");
00387     check(strcmp(xmpp_stanza_name(stanza), BIND) == 0,
00388           "Unexpected stanza");
00389 
00390     struct jid *new_jid = jid_new_from_jid(xmpp_client_jid(client));
00391 
00392     /* Jump to the inner <resource> tag. */
00393     stanza = xmpp_stanza_children(stanza);
00394     if (stanza == NULL || xmpp_stanza_data_length(stanza) == 0) {
00395         /* No resource specified, generate one. */
00396         char *uuid = make_uuid();
00397         jid_set_resource(new_jid, uuid);
00398         free(uuid);
00399     } else {
00400         check(strcmp(xmpp_stanza_uri(stanza), BIND_NS) == 0,
00401               "Unexpected stanza");
00402         check(strcmp(xmpp_stanza_name(stanza), RESOURCE) == 0,
00403               "Unexpected stanza");
00404 
00405         /* Copy the resource into the client information structure. */
00406         jid_set_resource(new_jid, xmpp_stanza_data(stanza));
00407     }
00408 
00409     /* Search for locally connected clients with this JID.  If a duplicate is
00410      * found, append a UUID until we get a unique resource. */
00411     while (xmpp_server_find_client(server, new_jid) != NULL) {
00412 
00413 #ifndef NDEBUG
00414         char *strjid = jid_to_str(new_jid);
00415         debug("JID %s is duplicate, generating new resource.", strjid);
00416         free(strjid);
00417 #endif
00418 
00419         /* UUID_SIZE includes the trailing null terminator. */
00420         char new_resource[strlen(jid_resource(new_jid)) + UUID_SIZE];
00421         char *uuid = make_uuid();
00422         snprintf(new_resource, JID_PART_MAX_LEN, "%s%s", jid_resource(new_jid),
00423                  uuid);
00424         free(uuid);
00425         jid_set_resource(new_jid, new_resource);
00426     }
00427 
00428     jid_set_resource(xmpp_client_jid(client), jid_resource(new_jid));
00429     jid_del(new_jid);
00430 
00431     char *jidstr = jid_to_str(xmpp_client_jid(client));
00432     utstring_printf(&success_msg, MSG_BIND_SUCCESS, id, jidstr);
00433     free(jidstr);
00434 
00435     /* Step 16: Server accepts submitted resourcepart and informs client of
00436      * successful resource binding */
00437     check(client_socket_sendall(xmpp_client_socket(client),
00438                                 utstring_body(&success_msg),
00439                                 utstring_len(&success_msg)) > 0,
00440           "Error sending resource binding success message.");
00441     utstring_done(&success_msg);
00442 
00443     /* Resource binding, and thus authentication, is complete!  Continue to
00444      * process general messages. */
00445     xmpp_parser_set_handler(parser, xmpp_core_handle_stanza);
00446     xmpp_server_add_stanza_route(server, xmpp_client_jid(client),
00447                                  xmpp_core_route_client, client);
00448     return true;
00449 
00450 error:
00451     utstring_done(&success_msg);
00452     return false;
00453 }
