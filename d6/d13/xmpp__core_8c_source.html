00044 bool xmpp_core_handle_stanza(struct xmpp_stanza *stanza,
00045                              struct xmpp_parser *parser, void *data) {
00046     struct xmpp_client *client = (struct xmpp_client*)data;
00047     struct xmpp_server *server = xmpp_client_server(client);
00048 
00049     const char *to = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TO);
00050     if (to == NULL) {
00051         /* RFC6120 Section 10, messages with no "to" are addressed to the bare
00052          * JID of the client, other stanzas are addressed to the server. */
00053         char *new_to;
00054         if (strcmp(xmpp_stanza_name(stanza), XMPP_STANZA_MESSAGE) == 0) {
00055             struct jid *bare = jid_new_from_jid_bare(xmpp_client_jid(client));
00056             new_to = jid_to_str(bare);
00057             jid_del(bare);
00058         } else {
00059             new_to = jid_to_str(xmpp_server_jid(server));
00060         }
00061         xmpp_stanza_set_attr(stanza, XMPP_STANZA_ATTR_TO, new_to);
00062     }
00063 
00064     /* RFC6120 Section 8.1.2.1, the server essentially ignores any "from"
00065      * attribute and adds the from attribute of the full JID of the client. */
00066     xmpp_stanza_set_attr(stanza, XMPP_STANZA_ATTR_FROM,
00067                          jid_to_str(xmpp_client_jid(client)));
00068 
00069     xmpp_server_route_stanza(server, stanza);
00070 
00071     return true;
00072 }

00075                             struct xmpp_server *server, void *data) {
00076     struct xmpp_client *client = (struct xmpp_client*)data;
00077 
00078     char *strjid = jid_to_str(xmpp_client_jid(client));
00079     debug("Routing to local client '%s'", strjid);
00080     free(strjid);

00082     size_t length;
00083     char *msg = xmpp_stanza_string(stanza, &length);
00084     if (client_socket_sendall(xmpp_client_socket(client), msg, length) <= 0) {
00085         free(msg);

00093 bool xmpp_core_route_server(struct xmpp_stanza *stanza,
00094                             struct xmpp_server *server, void *data) {
00095     if (strcmp(xmpp_stanza_name(stanza), XMPP_STANZA_MESSAGE) == 0) {
00096         log_warn("Message addressed to server?");

00098     } else if (strcmp(xmpp_stanza_name(stanza), XMPP_STANZA_PRESENCE) == 0) {
00099         log_warn("Ignoring presence stanza.");
