00236 static bool handle_message(struct xmpp_stanza *stanza, struct xep_muc *muc) {
00237     const char *type = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TYPE);
00238     check(type != NULL && strcmp(type, XMPP_STANZA_TYPE_GROUPCHAT) == 0,
00239           "MUC message stanza type other than groupchat.");
00240 
00241     char *to;
00242     STRDUP_CHECK(to, xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TO));
00243 
00244     struct jid *to_jid = jid_new_from_str(to);
00245     struct room *room = NULL;
00246     HASH_FIND_STR(muc->rooms, jid_local(to_jid), room);
00247     jid_del(to_jid);
00248     check(room != NULL, "MUC message to non-existent room.");
00249 
00250     char *from;
00251     STRDUP_CHECK(from, xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_FROM));
00252 
00253     struct jid *from_jid = jid_new_from_str(from);
00254     struct room_client *room_client = NULL;
00255     DL_FOREACH(room->clients, room_client) {
00256         if (jid_cmp(room_client->client_jid, from_jid) == 0) {
00257             break;
00258         }
00259     }
00260     jid_del(from_jid);
00261     check(room_client != NULL, "MUC message to unjoined room.");
00262 
00263     /* We need the "from" field of the stanza we send to be from the nickname
00264      * of the user who sent it. */
00265     struct jid *nick_jid = jid_new_from_jid(room->jid);
00266     jid_set_resource(nick_jid, room_client->nickname);
00267     char *nick_jid_str = jid_to_str(nick_jid);
00268     jid_del(nick_jid);
00269 
00270     xmpp_stanza_set_attr(stanza, XMPP_STANZA_ATTR_FROM, nick_jid_str);
00271 
00272     struct room_client *room_client_tmp;
00273     DL_FOREACH_SAFE(room->clients, room_client, room_client_tmp) {
00274         char *to_str = jid_to_str(room_client->client_jid);
00275         xmpp_stanza_set_attr(stanza, XMPP_STANZA_ATTR_TO, to_str);
00276         xmpp_server_route_stanza(muc->server, stanza);
00277     }
00278 
00279     /* Set the "to" and "from" back to their orignal values. */
00280     xmpp_stanza_set_attr(stanza, XMPP_STANZA_ATTR_TO, to);
00281     xmpp_stanza_set_attr(stanza, XMPP_STANZA_ATTR_FROM, from);
00282     return true;
00283 
00284 error:
00285     return false;
00286 }

00288 static bool handle_presence(struct xmpp_stanza *stanza, struct xep_muc *muc) {
00289     const char *to = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TO);
00290     check(to != NULL, "MUC message without to attribute.");
00291 
00292     struct jid *to_jid = jid_new_from_str(to);
00293     check(jid_resource(to_jid) != NULL, "MUC presence has no nickname");
00294     const char *search_room = jid_local(to_jid);
00295 
00296     bool rv;
00297     const char *type = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TYPE);
00298     if (type != NULL && strcmp(type, XMPP_STANZA_TYPE_UNAVAILABLE) == 0) {
00299         debug("Leaving room");
00300         rv = leave_room_presence(search_room, stanza, muc);
00301     } else {
00302         debug("Entering room");
00303         rv = enter_room_presence(search_room, stanza, muc);
00304     }
00305     jid_del(to_jid);
00306     return rv;
00307 
00308 error:
00309     return false;
00310 }

00335 static bool enter_room_presence(const char *search_room,
00336                                 struct xmpp_stanza *stanza,
00337                                 struct xep_muc *muc) {
00338     const char *to = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_TO);
00339     const char *from = xmpp_stanza_attr(stanza, XMPP_STANZA_ATTR_FROM);
00340 
00341     struct jid *to_jid = jid_new_from_str(to);
00342     struct jid *from_jid = jid_new_from_str(from);
00343 
00344     const char *nickname = jid_resource(to_jid);
00345 
00346     struct room *room = NULL;
00347     HASH_FIND_STR(muc->rooms, search_room, room);
00348     if (room == NULL) {
00349         /* Room doesn't exist, create it. */
00350         debug("New room, creating");
00351         room = room_new(muc, search_room);
00352         HASH_ADD_KEYPTR(hh, muc->rooms, room->name, strlen(room->name), room);
00353     } else {
00354         /* Room exists, check to make sure this nickname isn't used already. */
00355         struct room_client *room_client;
00356         DL_FOREACH(room->clients, room_client) {
00357             if (strncmp(nickname, room_client->nickname,
00358                         JID_PART_MAX_LEN) == 0) {
00359 
00360                 /* XEP-0045 Section 7.2.9 specifies that a client requesting to
00361                  * join a room with a duplicate nickname should be denied. */
00362                 debug("Duplicate nickname: %s", room_client->nickname);
00363                 /* "to" attribute is who we got the presence from, so reverse
00364                  * from/to. */
00365                 send_nickname_conflict(muc, room->jid, from, to);
00366 
00367                 goto done;
00368             }
00369         }
00370     }
00371 
00372     struct room_client *new_client = room_client_new(nickname, from_jid);
00373 
00374     /* If this is a local client, we need to know when they disconnect. */
00375     struct xmpp_client *client = xmpp_server_find_client(muc->server,
00376                                                          from_jid);
00377     if (client != NULL) {
00378         xmpp_server_add_client_listener(client, client_disconnect, muc);
00379     }
00380 
00381     send_presence_broadcast(muc, room, to, from, nickname);
00382 
00383     /* Add the client to the room client list. */
00384     DL_APPEND(room->clients, new_client);
00385 
00386 done:
00387     jid_del(to_jid);
00388     jid_del(from_jid);
00389     return true;
00390 }

00404         if (jid_cmp(room_client->client_jid, from_jid) == 0) {

00410     jid_del(from_jid);

00433 static bool leave_room(struct xep_muc *muc, struct room *room,
00434                        struct room_client *room_client) {
00435 
00436     jid_set_resource(room->jid, room_client->nickname);
00437 
00438     struct xmpp_stanza *presence = xmpp_stanza_new("presence", (const char*[]){
00439             XMPP_STANZA_ATTR_TYPE, "unavailable", NULL});
00440     xmpp_stanza_set_attr(presence, XMPP_STANZA_ATTR_FROM,
00441                          jid_to_str(room->jid));
00442     jid_set_resource(room->jid, NULL);
00443     xmpp_stanza_set_attr(presence, XMPP_STANZA_ATTR_TO,
00444                          jid_to_str(room_client->client_jid));
00445 
00446     struct xmpp_stanza *presence_x = xmpp_stanza_new("x", (const char*[]){
00447             "xmlns", "http://jabber.org/protocol/muc#user",
00448             NULL});
00449     xmpp_stanza_append_child(presence, presence_x);
00450     struct xmpp_stanza *presence_item = xmpp_stanza_new("item",
00451             (const char*[]){
00452             "affiliation", "member",
00453             "role", "none",
00454             NULL});
00455     xmpp_stanza_append_child(presence_x, presence_item);
00456 
00457     /* Send self-presence back to the occupant. */
00458     struct xmpp_stanza *status = xmpp_stanza_new("status",
00459         (const char*[]){"code", "110", NULL});
00460     xmpp_stanza_append_child(presence_x, status);
00461 
00462     xmpp_server_route_stanza(muc->server, presence);
00463     DL_DELETE(room->clients, room_client);
00464     room_client_del(room_client);
00465 
00466     xmpp_stanza_remove_child(presence_x, status);
00467     xmpp_stanza_del(status, true);
00468 
00469     /* Send the leave presence to other occupants in the room. */
00470     struct room_client *other_client;
00471     DL_FOREACH(room->clients, other_client) {
00472         xmpp_stanza_set_attr(presence, XMPP_STANZA_ATTR_TO,
00473                              jid_to_str(other_client->client_jid));
00474         xmpp_server_route_stanza(muc->server, presence);
00475     }
00476     xmpp_stanza_del(presence, true);
00477 
00478     /* If the room is empty now, delete it. */
00479     if (room->clients == NULL) {
00480         HASH_DEL(muc->rooms, room);
00481         room_del(room);
00482     }
00483     return true;
00484 }
