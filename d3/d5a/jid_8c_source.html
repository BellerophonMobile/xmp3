00066 struct jid* jid_new_from_str(const char *jidstr) {
00067     struct jid *jid = jid_new();
00068 
00069     const char *at_delim = strchr(jidstr, '@');
00070     const char *slash_delim = strchr(jidstr, '/');
00071 
00072     if (at_delim != NULL) {
00073         int len = at_delim - jidstr;
00074         if (len > JID_PART_MAX_LEN) {
00075             len = JID_PART_MAX_LEN;
00076         }
00077         check(len > 0, "JID local part cannot be empty.");
00078         STRNDUP_CHECK(jid->local, jidstr, len);
00079         /* This will catch if '@' and '/' are out of order, since the len check
00080          * below will be negative. */
00081         jidstr = at_delim + 1;
00082     }
00083 
00084     if (slash_delim != NULL) {
00085         int len = slash_delim - jidstr;
00086         if (len > JID_PART_MAX_LEN) {
00087             len = JID_PART_MAX_LEN;
00088         }
00089         check(len > 0, "JID domain part cannot be empty.");
00090         STRNDUP_CHECK(jid->domain, jidstr, len);
00091 
00092         check(slash_delim[1] != '\0', "JID resource part cannot be empty.");
00093         STRNDUP_CHECK(jid->resource, slash_delim + 1, JID_PART_MAX_LEN);
00094     } else {
00095         check(jidstr[0] != '\0', "JID domain part cannot be empty.");
00096         STRNDUP_CHECK(jid->domain, jidstr, JID_PART_MAX_LEN);
00097     }
00098     return jid;
00099 
00100 error:
00101     jid_del(jid);
00102     return NULL;
00103 }

00122 char* jid_to_str(const struct jid *jid) {
00123     /* Domain part is required */
00124     if (jid->domain == NULL) {
00125         return NULL;
00126     }
00127 
00128     UT_string strjid;
00129     utstring_init(&strjid);
00130 
00131     if (jid->local == NULL) {
00132         utstring_printf(&strjid, jid->domain);
00133     } else {
00134         utstring_printf(&strjid, "%s@%s", jid->local, jid->domain);
00135     }
00136 
00137     if (jid->resource != NULL) {
00138         utstring_printf(&strjid, "/%s", jid->resource);
00139     }
00140 
00141     /* Explicitly using utstring_init instead of utstring_new cleans up the
00142      * UT_string structure when the function exits.  Since we never call
00143      * utstring_done or utstring_free, the actual string being created inside
00144      * the structure is never freed, so this does not leak memory. */
00145     return utstring_body(&strjid);

00197 int jid_cmp_wildcards(const struct jid *a, const struct jid *b) {
00198     if (a->local == NULL) {

00221                 != 0) {

00227                 != 0) {
